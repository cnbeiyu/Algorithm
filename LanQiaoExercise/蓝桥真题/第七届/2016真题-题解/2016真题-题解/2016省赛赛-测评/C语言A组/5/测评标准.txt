x & (x+1)

正确直接得分，否则把用户答案带入测评代码中的######，运行结果比对：
00000000000000000000000011000010....00000000000000000000000011000010
00000000000000000000000011001111....00000000000000000000000011000000
00000000000000000000000011000001....00000000000000000000000011000000


测评程序为：

#include <stdio.h>

void f(int x)
{
	int i;
	for(i=0; i<32; i++) printf("%d", (x>>(31-i))&1);
	printf("....");
	
	x = ######;
	
	for(i=0; i<32; i++) printf("%d", (x>>(31-i))&1);
	printf("\n");	
}

int main()
{
	f(128+64+2);
	f(128+64+15);
	f(128+64+1);
	
	return 0;
}
